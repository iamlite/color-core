import { useState } from 'react';
import { Color } from 'color-core';

# Getting Started with color-core

Let's get you up and running with color-core quickly.

## Installation

Install color-core using npm:

```bash
npm install color-core
```

## Basic Usage

First, let's create a new Color object:

```typescript
import { Color } from 'color-core';

const purple = new Color('#B05BFF');
```

Now, let's explore what we can do with it:

export const ColorSwatch = ({ color }) => (
  <div className="color-swatch" style={{ backgroundColor: color }} />
);

export const ColorDemo = () => {
  const [color, setColor] = useState(new Color('#B05BFF'));

  return (
    <div className="color-demo">
      <p className="color-instruction">Click the color below to change it:</p>
      <input
        type="color"
        value={color.toHex()}
        onChange={(e) => setColor(new Color(e.target.value))}
      />
      <p>Hex: {color.toHex()}</p>
    </div>
  );
};

<ColorDemo />

### Converting to RGB

You can easily convert the color to RGB format:

```typescript
const rgbColor = purple.toRgb();
```

export const RgbDemo = () => {
  const [color, setColor] = useState(new Color('#B05BFF'));

  return (
    <div className="color-demo">
      <p className="color-instruction">Try changing the color:</p>
      <input
        type="color"
        value={color.toHex()}
        onChange={(e) => setColor(new Color(e.target.value))}
      />
      <p>RGB: {JSON.stringify(color.toRgb())}</p>
    </div>
  );
};

<RgbDemo />

### Adjusting Lightness

You can make the color lighter or darker:

```typescript
const lightpurple = purple.adjustLightness(10);
```

export const LightnessDemo = () => {
  const [color, setColor] = useState(new Color('#B05BFF'));

  const lighten = () => {
    setColor(color.adjustLightness(10));
  };

  const reset = () => {
    setColor(new Color('#B05BFF'));
  };

  return (
    <div className="color-demo">
      <ColorSwatch color={color.toHex()} />
      <p>Current color: {color.toHex()}</p>
      <button onClick={lighten}>Click to Lighten</button>
      <button onClick={reset}>Click to Reset</button>
    </div>
  );
};

<LightnessDemo />

### Generating Color Harmonies

You can generate a triadic color scheme:

```typescript
const [color1, color2, color3] = purple.triadic();
```

export const TriadicDemo = () => {
  const [color, setColor] = useState(new Color('#B05BFF'));
  const triadicColors = color.triadic();

  return (
    <div className="color-demo">
      <p className="color-instruction">Change the base color to see triadic harmony:</p>
      <input
        type="color"
        value={color.toHex()}
        onChange={(e) => setColor(new Color(e.target.value))}
      />
      <div className="swatch-container">
        <ColorSwatch color={triadicColors[0].toHex()} />
        <ColorSwatch color={triadicColors[1].toHex()} />
        <ColorSwatch color={triadicColors[2].toHex()} />
      </div>
      <p>Triadic colors: {triadicColors[0].toHex()}, {triadicColors[1].toHex()}, {triadicColors[2].toHex()}</p>
    </div>
  );
};

<TriadicDemo />

That's it! You've now created colors, converted between formats, manipulated colors, and generated color harmonies using color-core.
